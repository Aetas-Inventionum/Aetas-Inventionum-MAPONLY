african_terminus_disunity = {
	potential = {
		owner = {
			NOT = {
				AND = {
					owns_core_province = 1751
					owns_core_province = 334
				}
			}
		}
	}

	trigger = {
		hidden_trigger = {
			owner = {
				NOT = {
					AND = {
						owns_core_province = 1751
						owns_core_province = 334
					}
				}
			}
		}
	}

	province_trade_power_modifier = -0.2
	trade_goods_size = -0.5
	local_center_of_trade_upgrade_cost = 0.33
	
	#Province scope
	on_activation = {
		1751 = {
  			owner = {
    			country_event = {
      				id = aetas_trade_events.2
 				}
  			}
		}
		334 = {
  			owner = {
    			country_event = {
      				id = aetas_trade_events.2
 				}
  			}
		}
	}
	
	on_deactivation = {
		1751 = {
  			owner = {
    			country_event = {
      				id = aetas_trade_events.1
 				}
  			}
		}
	}	
}
