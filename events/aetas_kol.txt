						religion = protestant

					}

					who = ROOT

				}
			}

				limit = { 
					is_part_of_hre = yes

				}

#archbishop seizes power and secularizes the bishopric
country_event = {
	id = aetas_kol.10
	title = aetas_kol.10.t
	desc = aetas_kol.10.d
	picture = PRINCE_BISHOP_RESIGN_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes 
	major = yes 
	major_trigger = {
		is_part_of_hre = yes 
	}

	option = { 
		name = aetas_kol.10.a
		if = {
			limit = { 
				num_of_religion = {
					religion = protestant
					value = 0.5
				}
			}
			change_religion = protestant 
		}
		else_if = { 
			limit = { 
				num_of_religion = {
					religion = reformed 
					value = 0.5
				}
			}
			change_religion = reformed 
		}
		else_if = {
			limit = { 
				num_of_religion = {
					religion = hussite
					value = 0.5
				}
			}
			change_religion = hussite 
		}
		else = { change_religion = reformed } #reformed is backup if there is no majority as it was the religion of choice historically by Gebhard Truchseß von Waldburg
		change_government = monarchy
		set_government_rank = 1
		add_government_reform = feudalism_reform
		add_government_reform = truchsessian_reform
		define_consort = { 
			name = "Agnes"
			dynasty = "von Mansfeld-Eisleben"
			religion = reformed
			adm = 5
			dip = 5
			mil = 1
			female = yes 
			age = 32
		}
		define_advisor = { 
			type = theologian 
			name = "Agnes von Mansfeld-Eisleben"
			skill = 3
			religion = reformed 
			female = yes 
			discount = yes 
		}
		set_imperial_incident = incident_colognian_war
		hidden_effect = { 
			remove_historical_rival = KLE 
			add_historical_friend = KLE 
			add_historical_friend = BRG 
			add_historical_friend = JUL 
			add_historical_friend = JCB
		}
	}
}
