#we have captured book of kells 
country_event = {
	id = aetas_ire.3
	title = aetas_ire.3.t
	desc = aetas_ire.3.d
	picture = BIG_BOOK_eventPicture
	
	is_triggered_only = yes 
	major = yes 
	major_trigger = { 
		culture_group = gaelic
	}

	trigger = {
		primary_culture = irish 
		FROM = {
			has_province_modifier = book_of_kells
			NOT = {
				owned_by = ROOT
			}
			NOT = {
				owner = {
					OR = {
						is_in_war = {
							defenders = PREV
							defenders = ROOT
						}
						is_in_war = {
							attackers = PREV
							attackers = ROOT
						}
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				save_global_event_target_as = book_of_kells_province
				remove_province_modifier = book_of_kells
			}
		}
	}
	
	option = {
		name = aetas_ire.3.a
		add_prestige = 10
		hidden_effect = {
			FROM = {
				owner = {
					country_event = {
						id = aetas_ire.4
					}
				}
			}
		}
		capital_scope = {
			add_province_triggered_modifier = book_of_kells
		}
	}
}
