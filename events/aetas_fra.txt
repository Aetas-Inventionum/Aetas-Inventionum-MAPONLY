# Death of the Duke of Berry
country_event = {
	id = aetas_fra.1
	title = aetas_fra.1.t
	desc = aetas_fra.1.d
	picture = GOOD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	major = yes 
	major_trigger = { 
		is_neighbor_of = ROOT
	}

	trigger = { 
		tag = FRA 
		is_year = 1416
		BER = { is_subject_of = ROOT }
	}

	option = { #oops!
		name = "aetas_fra.1.a"
		set_country_flag = duc_berry_inheritance
		massif_central_area = { 
			limit = { owned_by = BER }
			cede_province = BOU 
			add_core = BOU
			remove_core = BER 
		}
		if = { 
			limit = { 3216 = { owned_by = BER } }
			3216 = { cede_province = SNC }
			SNC = { 
				define_ruler = { 
					name = "Jean V"
					dynasty = "de Bueil"
					age = 15
					adm = 2
					dip = 2
					mil = 4
				}
			}
			create_subject = { 
				who = SNC 
				subject_type = fief 
			}
			SNC = { 
				join_all_offensive_wars_of = ROOT 
				join_all_defensive_wars_of = ROOT
			}
		}

		if = { 
			limit = { 
				exists = TRN 
				TRN = { has_ruler = "Jean I" }
			}
			TRN = { inherit = BER }
		}
		else = { inherit = BER }
		hidden_effect = { 
			every_province = { 
				limit = { 
					area = poitou_area 
					is_core = BER 
				}
				remove_core = BER
			}
		}
	}
}
