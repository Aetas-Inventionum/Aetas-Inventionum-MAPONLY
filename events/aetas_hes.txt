country_event = {
	id = aetas_hes.4
	title = aetas_hes.4.t
	desc = aetas_hes.4.d
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	major = yes 
	major_trigger = { 
		OR = { 
			tag = HES
			tag = KSS
		}
	}

	trigger = { 
		tag = KTZ
		has_heir = no 
		OR = {
			has_ruler = "Johann IV"
			has_ruler = "Philipp I"
		}
		OR = { 
			alliance_with = HES 
			alliance_with = KSS 
		}
		any_country = { mission_completed = hes_ktz_ally }
	}

	mean_time_to_happen = { 
		months = 120
	}

	option = { 
		name = aetas_hes.4.a #let hesse inherit
		ai_chance = { 
			factor = 1
			modifier = { 
				factor = 100
				OR = { 
					HES = { ai = no }
					KSS = { ai = no }
				}
			}
		}
		set_country_flag = hes_inherit_ktz_flag
		custom_tooltip = hes_inherit_ktz.tt
		hidden_effect = { 
			if = { 
				limit = { exists = HES }
				HES = { country_event = { id = aetas_hes.5 } }
			}
			else = { 
				KSS = { country_event = { id = aetas_hes.5 } }
			}
		}
	}
	option = { 
		name = aetas_hes.4.b #dont!
		ai_chance = { 
			factor = 2 
			modifier = { 
				factor = 0
				OR = { 
					HES = { ai = no }
					KSS = { ai = no }
				}
			}
		}
		if = { 
			limit = { exists = HES }
			HES = {
				add_casus_belli = {
					target = KTZ
					type = cb_restore_personal_union
					months = 480
				}
			}
		}
		else = { 
			KSS = {
				add_casus_belli = {
					target = KTZ
					type = cb_restore_personal_union
					months = 480
				}
			}
		}
	}
}
