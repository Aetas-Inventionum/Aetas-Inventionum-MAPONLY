#Naming a new Duke of Maine
country_event = {
	id = aetas_hyw.5
	title = aetas_hyw.5.t
	desc = aetas_hyw.5.d
	picture = CONQUEST_eventPicture
	
	fire_only_once = yes
	major = yes 
	major_trigger = { 
		defensive_war_with = ROOT 
	}

	trigger = {
		tag = ENG 
		has_global_flag = hyw_flag
		has_country_flag = eng_nrm_flag
		all_province = {
			OR = { 
				province_id = 3202
				province_id = 3204
				province_id = 3203
				province_id = 177
			}
			controller = { 
				OR = { 
					tag = ENG 
					alliance_with = ENG 
					is_subject_of = ENG 
				}
			}
		}
	}

	option = { 
		name = aetas_hyw.5.a #Have noble John of Lancaster, Duke of Bedford, become Duke of Maine.
		ai_chance = { factor = 5 } #historical
		every_province = { 
			limit = { 
				OR = { 
					province_id = 3202
					province_id = 3204
					province_id = 3203
					province_id = 177
				}
			}
			cede_province = MNE 
		}
		create_subject = { 
			who = MNE 
			subject_type = appanage 
		}
		MNE = { 
			add_government_reform = appanage_reform
			define_ruler = { 
				name = "John I"
				dynasty = "of Lancaster"
				culture = english 
				age = 35
				adm = 5
				dip = 4
				mil = 4
			}
			define_consort = { 
				name = "Anne"
				dynasty = "de Valois-Bourgogne"
				culture = burgundian 
				country_of_origin = BUR 
				age = 20
				adm = 1
				dip = 3
				mil = 1
			}
			capital_scope = { 
				build_to_forcelimit = { 
					infantry = 0.8
					cavalry = 0.2
				}
			}
			join_all_offensive_wars_of = ROOT 
		}
		set_global_flag = eng_mne_john_flag
		hidden_effect = { 
			MNE = { change_country_color = { color = { 200 40 36 } } }
		}
	}
	option = { 
		name = aetas_hyw.5.b #it would be best to just rule it directly
		every_province = { 
			limit = { 
				OR = { 
					province_id = 3202
					province_id = 3204
					province_id = 3203
					province_id = 177
				}
			}
			cede_province = ROOT 
			add_core = ROOT 
			add_local_autonomy = 25
		}
	}
	after = { 
		hidden_effect = { set_global_flag = eng_mne_flag }
	}
}

	major = yes

	}

	trigger = {
		tag = ANJ
		has_global_flag = hyw_flag
		has_global_flag = eng_mne_flag
		all_province = {
			OR = { 
				province_id = 3203
				province_id = 177
			}
			controller = { 
				OR = { 
					tag = ANJ  
					alliance_with = ANJ  
					is_subject_of = ANJ 
					tag = FRA 
					alliance_with = FRA 
					is_subject_of = FRA 
				}
			}
		}
	}

	option = { 
		name = aetas_hyw.6.a #Alright, you may have it, Charles.
		ai_chance = { factor = 10 } #historical 
		if = { 
			limit = { 
				AND = { 
					exists = MNE 
					has_global_flag = eng_mne_john_flag
				}
			}
			inherit = MNE
			hidden_effect = { MNE = { restore_country_color = yes } }
		}
		every_owned_province = { 
			limit = { is_core = MNE }
			cede_province = MNE 
			remove_core = ROOT 
		}
		create_subject = { 
			who = MNE 
			subject_type = appanage
		}
		MNE = { 
			add_government_reform = appanage_reform
			define_ruler = {
				name = "Charles IV"
				dynasty = "de Valois"
				age = 34
				adm = 2
				dip = 2

			}
			capital_scope = { 
				build_to_forcelimit = { 
					infantry = 0.8
					cavalry = 0.2
				}
			}
			join_all_offensive_wars_of = ROOT 
		}
	}
	option = { 
		name = aetas_hyw.6.b #Now  now, Charles!
		ai_chance = { factor = 5 }
		every_province = { 
			limit = { 
				OR = { 
					province_id = 3203
					province_id = 177
				}
			}
			cede_province = ROOT 
			add_core = ROOT 
		}
	}
