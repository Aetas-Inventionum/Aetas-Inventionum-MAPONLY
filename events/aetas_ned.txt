	desc = aetas_ned.10.d
	picture = REFUGEES_ESCAPING_eventPicture
	
	mean_time_to_happen = {
		months = 40
	} 

	trigger =  {
		OR = {
			tag = NED 
			tag = PRU
		}
		has_country_flag = ned_immigrants #no modifier
		OR = { 
			current_age = age_of_reformation 
			current_age = age_of_absolutism
		}
	}
	
		option = { #we love random immigrants
			name = "aetas_ned.10.a"
			random_owned_province = { 
				limit = { 
					NOT = { has_province_modifier = ned_immigrants_welcomed }
				}
				add_province_modifier = { 
					name = "ned_immigrants_welcomed"
					duration = 3650
				}
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
}

#cheat event to form ned
country_event = {
	id = aetas_ned.100 
	title = aetas_ned.100.t
	desc = aetas_ned.100.d
	picture = REMBRANDT_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	option = { 
		name = aetas_ned.100.a 
		if = {
			limit = {
				is_elector = no
				is_emperor = no
				OR = {
					is_part_of_hre = yes
					any_owned_province = {
						is_part_of_hre = yes
					}
				}
			}
			set_in_empire = no
			hidden_effect = {
				every_owned_province = {
					limit = {
						is_part_of_hre = yes
					}
					set_in_empire = no
				}
			}
		}
		hidden_effect = {
			inherit = UTR 
			inherit = GRN 
			inherit = GEL 
			inherit = ZEE 
			inherit = ZUT 
			inherit = FRI 
			inherit = OMM 
		}
		change_tag = NED 
		change_government = republic
		add_government_reform = dutch_republic
		if = {
			limit = {
				has_custom_ideas = no
				NOT = { has_idea_group = ai_NED_ideas }
			}
			country_event = { id = ideagroups.1 } #Swap Ideas
		}
		swap_non_generic_missions = yes
		set_government_rank = 2
	}
}
