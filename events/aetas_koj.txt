	fire_only_once = yes

		name = aetas_koj.8.a #union with the kingdom
		egypt_region = { 
			limit = { country_or_non_sovereign_subject_holds = ROOT }
			add_core = KEM
			cede_province = KEM 
		}
		sinai_area = { 
			limit = { 
				country_or_non_sovereign_subject_holds = ROOT 
				NOT = { province_id = 4268 }
			}
			add_core = KEM 
			cede_province = KEM 
		}
		KEM = { 
			set_capital = 363
		}
		363 = { 
			change_culture = KEM 
			change_religion = KEM 
		}
		every_province = { 
			limit = { culture = kemi }
			add_core = KEM
		}
		every_province = { 
			limit = { region = egypt_region }
			add_permanent_claim = KEM 
		}
		create_union = KEM 
	}
	option = { 
		name = aetas_koj.8.b #no thanks
		add_prestige = -10
	}
	option = {
		name = aetas_koj.8.c #become the kingdom
		change_tag = KEM 
		change_primary_culture = kemi 
		add_accepted_culture = hierosolymitan
		363 = { 
			if = { 
				limit = { NOT = { is_core = ROOT } }
				add_core = ROOT  
			}
			if = { 
				limit = { NOT = { owned_by = ROOT } }
				cede_province = ROOT
			}
		}
		set_capital = 363 
		363 = { 
			change_culture = ROOT 
			change_religion = ROOT 
		}
		every_province = { 
			limit = { culture = kemi }
			add_core = ROOT
		}
		egypt_region = {
			limit = {
				NOT = { is_permanent_claim = ROOT } 
				NOT = { is_core = ROOT } 
			}
			add_permanent_claim = ROOT
		}
	}
}
